Topological sorting for Directed Acyclic Graph (DAG) is a linear ordering of vertices such that for every directed edge u v, vertex u comes before v in the ordering. Topological Sorting for a graph is not possible if the graph is not a DAG. - GeekForGeeks 
            int n; // number of vertices
            vector> adj; // adjacency list of graph
            vector visited;
            vector ans;

            void dfs(int v) {
                visited[v] = true;
                for (int u : adj[v]) {
                    if (!visited[u])
                        dfs(u);
                }
                ans.push_back(v);
            }

            void topological_sort() {
                visited.assign(n, false);
                ans.clear();
                for (int i = 0; i < n; ++i) {
                    if (!visited[i])
                        dfs(i);
                }
                reverse(ans.begin(), ans.end());
}
